<?php
# echo可以用逗号分隔要输出的内容

//echo 不是一个函数 而是一个语言固有的结构 

$a = 1;
$b = "b";
echo $a,", ",$b,"<br/>";  //1, b

//面试题 哪个语句执行最快

echo "hi " . "yutou";  //双引号比单引号慢
echo 'hi ' . 'yutou';  //要先进行字符串连接运算 再输出 较慢
echo 'hi ' , 'yutou';  //这个最快
print_r('hi ' , 'yutou');  //这个需要调用函数 比语言结构慢

 

# 变量的引用赋值

$a = &$b;  //不是$b指向$a 而是它俩指向同一内存地址
echo $a,", ",$b,"<br/>";  //b, b
$b = 5;
echo $a,", ",$b,"<br/>";  //5, 5

关于unset

unset($a);    //unset掉了a 只是a不要了 这个内存地址还被b所指向 所以并未被销毁 b也unset掉了 内存的存储才被垃圾回收机制释放
echo $a,", ",$b,"<br/>";  //, 5

 

# 引用计数赋值

class human {
    public $age = 21;
}
$xiuxiu = new human();  //创建对象 其实$xiuxiu指向一个内存地址1 这里面存的是一个指向对象的内存地址2和引用计数 目前是1

$yutou = $xiuxiu;  //对象的赋值不要加& 它自动就是引用赋值 是$yutou和$xiuxiu都指向内存地址1 并且引用计数变为2

$yutou->age = 20;  //两者指向一块区域 修改这个 就是修改那个
print_r($xiuxiu);  //human Object ( [age] => 20 ) 
echo "<br/>";

$yutou = 999;  //把内存地址1中存的引用计数变成了1 再开辟一个新内存存放999让$yutou指向它

（这个是老师画的 不是我画的 赫赫）

print_r($yutou);  //999
echo "<br/>";
print_r($xiuxiu);  //human Object ( [age] => 20 ) 
echo "<br/>";

# 动态变量名

//php中变量名 函数名 对象名 类名 都很灵活 可以用变量表示
//这样可以接收用户的参数 更灵活

$a = "a";
$b = "b";
$c = "c";

$who = "b";  //这个可以是用户的传参
echo $$who,"<br/>";  //b  -  真神奇

$shui = "who";
echo $$$shui,"<br/>";  //b  -  太神奇了吧

# 运算符

echo (3 / 2),"<br/>";  //1.5  -  弱类型 如果是浮点直接返回浮点
echo "(0 == \"0\"): ",(0 == "0"),"<br/>";  //1
echo "(0 == \"null\"): ",(0 == "null"),"<br/>";  //1
echo "(null == \"\"): ",(null == ""),"<br/>";  //1
echo "(null == \"null\"): ",(null == "null"),"<br/>";  //[没有输出]  -  "null"是字符串 不是空的 所以不等
echo "(0 == null): ",(0 == null),"<br/>";  //1
echo "(0 === null): ",(0 === null),"<br/>";  //[没有输出]  -  全等于 不仅相等 类型也得一样

//面试题 举例说明用===可以判断 用==不能判断的

if (false !== strpos('echo', 'e')) {  //这里必须用全不等于判断 因为strpos函数 返回参数2在参数1中的位置 如果不存在返回false 它恰好在第0位 如果用!=判断 0也是false 顾得不到想要的结果
    echo '有e';
} else {
    echo '没有e';
}
echo "<br/>";

# 函数

// 带默认值得参数

function func1($a, $b = 2) {
    echo $a + $b;
}
echo func1(5);  //7  -  有默认值的参数可以不传 如果有默认值的参数不在最后面 就必须得传

echo func1(5, 6, 7);  //11  -  传参可以多余函数定义形参的个数 多余的被忽略 但不能少于 会报错

// 函数引用传值方式（严重不推荐这样写 违反了封装原则）

$a = 10;
function dbl(&$x) {
    $x *= 2;
    return $x;
}
echo dbl($a),"<br/>";  //20
echo $a,"<br/>";  //20

// 函数内部操作全局变量（严重不推荐使用）

/*直接定义在php文件中的变量就是全局变量*/
$a = 1;
function func2() {
    echo $a,"<br/>";  //这个$a 是在函数内部定义的 和外边的不一样 所以这不是访问外边的$a
    global $a;  //告诉函数 这是全局变量
    $a = 2;
    echo $a,"<br/>";
}
func2();
echo $a,"<br/>";


// 动态调用函数

function a($i) { echo "a $i<br/>"; }
function b($i) { echo "b $i<br/>"; }
function c($i) { echo "c $i<br/>"; }

// 方式1
$func = "b";  //指定要调用的函数名
$func(3);  //这就动态调用了  -  太神妙了吧

// 方式2
call_user_func("a", 1);
call_user_func_array("c", array(4));  //把所有参数组织成数组传递

 

# 超级全局变量

在所有部分都可见 无论是函数中（在函数中不用global声明） 还是数组中

$_GET, $POST, $_COOKIE, $_SESSION, $_SERVER, $_ENV, $_REQUEST, $_FILE, $GLOBALS

$_GET, $POST  -  请求参数
$_SERVER  -  服务器执行时的相关信息 和服务器软件的不同会略有不同 来访者ip 请求参数 URI 文件绝对路径
$_ENV

echo $_ENV,"<br/>";  //可能为空 因为php.ini中限制了对它的打印

php.ini

variables_order = "GPCS"   //限制可以打印 GET POST COOKIE SESSION 这个顺序还决定了他们的优先级
variables_order = "GPCSE"  //要是想打印的话在后面加上E 代表ENV 但是服务器一般限制打印 不让得到过多的服务器信息

但是还可以通过 getenv() 函数得到

例如

echo getenv('OS'),"<br/>";  //得到操作系统

因为各种操作系统间有很多不一样的地方
就比如说换行符

windows "\r\n"
linux "\n"
mac "\r"

# 超级全局变量 $_GET, $_POST

它俩是数组类型的变量

关于参数大小 面试

GET方式是根据URL传递 URL的长度受到 浏览器 和 服务器软件 的双重制约
具体长度不一定 在2K以内 不要说就是2K 一看就是刚学出来的 其实HTTP协议没有限制

POST当时主要是受到 服务器软件的限制

php.ini 中可以配置 

upload_max_filesize = 2M  //默认为2M

 

# 超全局变量 $GLOBALE 数组

是所有全局变量的花名册 里边有 声明的全局变量 在这里都能找到

给在函数内部操作全局变量（不推荐使用） 提供了方法

操作$GLOBALS中的某个成员的值 就等于在操作那个全局变量

$a = 99;
print_r($GLOBALS);
/* 结果
...
[a] => 99
...
*/

$b = 100;
print_r($GLOBALS);
/* 结果
...
[a] => 99
[b] => 100
...
*/
