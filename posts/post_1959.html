$_CFG['username'] = 'root';

$_CFG['password'] = 'root';

$_CFG['database'] = 'tieba';

$_CFG['charset'] = 'utf8';

 

define('DEBUG',true);

2.初始化文件

init.php

define('ROOT', );

require(ROOT . '/config.inc.php');

require(ROOT . '/mysql.func.php');

header('Content-type:text/html; charset=utf-8');

if (DEBUG) {

error_reporting(E_ALL);

} else {

error_reporting(0);

}

if (!get_magic_quotes_gpc()) {

function _addslashes(&$value, $key) {

$value = addslashes($value);

}

array_walk_recursive(&$_POST, '_addslashes');

array_walk_recursive(&$_GET, '_addslashes');

array_walk_recursive(&$_COOKIE, '_addslashes');

}

$conn = mysql_conn($_CFG['host'], $_CFG['username'], $_CFG['password']);

if (!$conn) {

exit('数据库连接出错...');

}

$sql = 'use ' . $_CFG['database'];

mysql_query($sql, $conn);
