  public class Reflect {

public static void main(String[] args) throws Exception {

/* 找到类 */ String className = "cn.yutouxiuxiu.study.Student"; //1.通过类的全限定名称获得。常用于框架中 Class clazz = Class.forName(className);

/* 实例化 */ Constructor con = clazz.getConstructor(int.class); //2.已知类，用来确定方法参数列表 在getConstructor时 Object obj = con.newInstance(1);

System.out.println(obj.getClass()); //3.已知对象，方法体获得运行时类型

/* 执行方法 */ Method method = clazz.getMethod("method", int.class); //获取方法 method.invoke(obj, 1); //obj为该类的实例

/* 获取字段 */ Field field = clazz.getField("name"); System.out.println(field.get(obj));

/* 执行私有方法 */ Method privateMethod = clazz.getDeclaredMethod("privateMethod", null); //获取定义的方法（包括私有方法） privateMethod.setAccessible(true); //设置可进入性 privateMethod.invoke(obj, null);

}

}   public class Student {

public String name = "yutou";

static {

System.out.println("类加载进内存");

}

System.out.println("默认构造");

}

public Student(int a) {

System.out.println("构造" + a);

}

public void method() {

System.out.println("方法");

}

public void method(int a) {

System.out.println("方法" + a);

}

private void privateMethod() {

System.out.println("私有方法");

}

}
