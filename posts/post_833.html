用户自动登录 过滤器

{ post:time }
零、原理

让用户无论访问哪个页面 都是自动登录上的 所以用到过滤器

 

一、实现

正常登录时

在正常登录的时候 如果勾选了自动登录

返回一个Cookie信息存在浏览器

Cookie c = new Cookie(“auto-login”, username + “,”+ password);

c.setMaxAge(…);

c.setPath(…);

自动登录时

用Filter来处理自动登录

public void doFilter(ServletRequest request, ServletResponse response,
        FilterChain chain) throws IOException, ServletException {
   
    HttpServletRequest httpRequest = (HttpServletRequest) request;
    HttpServletResponse httpResponse = (HttpServletResponse) response;
   
    //1.只给未登录的用户 才需要自动登录
    if (httpRequest.getSession(false) == null || httpRequest.getSession().getAttribute("user") == null) {
       
        //2.只给带了auto-login的Cookie信息的才给自动登录
        Cookie[] cookies = httpRequest.getCookies();
        Cookie cookie = null;
        if (cookies != null) {
            for (Cookie c : cookies) {
                if ("auto-login".equals(c.getName())) {
                    cookie = c;
                   
                }
            }
        }
       
        if (cookie != null) {
            String username = cookie.getValue().split(",")[0];
            String password = cookie.getValue().split(",")[1];
        }
        //3.只给Cookie信息中用户名密码对的自动登录

        }
    }
    chain.doFilter(request, response);    //不用传httpResponse 传respnse就行 httpResponse和response都是引用 本质是一个
}
