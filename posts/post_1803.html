1.Array

（1）定义数组

var arr1 = [];

var arr2 = new Array();

arr1.push("a");  //添加内容

arr1.push("b");

alert(arr1.join(","));  //用“,”分割输出内容

2.String

"".substr(m, n);

"".substring(m, n);

3.RegExp

var regex = /^\d+$/;

alert(regex.test("12345"));

4.Functions

（1）定义函数

function a() { alert(1); }

a();  //输出2 它是先编译函数

function a() { alert(2); }

a();  //输出2

var b = function() { alert(1); };

b();  //输出1

b = function() { alert(2); };

b();  //输出2

var c = new Function(arg1, arg2, ..., "alert(1);");

c(...);

五、DOM对象

1.获取内容

document.getElementById('').value();

document.getElementById('').innerHTML();

要写在DOM元素的下面 因为是按顺序执行的 要不然就放在onload里

window.onload = function() {};

 

2.向select添加option

（1）XMLDOM方法（通用的 重点学）

var option = document.createElement("option");

option.innerHTML = "yutou";

document.getElementById("").appendChild(option);

（2）HTMLDOM方法

var option = new Option();

option.text = "yutou";

document.getElementById("").add(option);

3.加属性

document.getElementById("").setAttribute("", "");

4.添加style

document.getElementById("").style.color = "red";

 

六、javascript的高级课程

1.创建对象

对象结构 就是Map

var obj = {};  //Object类型

var obj = {

name : "";

getName : function() {

return this.name;

}

};

alert(obj.getName());

类结构 函数的对象

var Person = function() {

this.age = 20;

};

var p = new Person();

alert(p.age);

2.JSON可以转为对象

var json = '{"name":"芋头","age":"20"}';

var obj = eval("(" + json + ")");

alert(obj.name);
