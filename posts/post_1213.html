全局域和函数域

定义局部变量没有var修饰时

 

var a = “a”;  //全局变量 在环境的任意位置都可以访问

var d = “d”;

function fn() {

    var b = “b”;  // 局部变量 在对应的环境中访问

    var d = “d”;  //调用函数时执行至此 初始化变量d

    alert(d);

    alert(a);

    alert(c);

    alert(b);

}

//程序执行至此 会定义出局部变量d 并不初始化

1.匿名函数

没有函数名的函数

function() {

}

页面加载时 有时需要初始化工作

做页面初始化的函数只用一次 就可以用匿名函数

2.回调函数

可以将匿名函数作为参数传递给其他函数

将one, two作为参数传递给fn() one,two函数就叫做回调函数

    回调函数：将一个函数作为另一个函数的参数传递 作为参数的函数叫回调函数

function fn(a, b) {

    return a() + b();

}

//var one = function(){ return 1; }

//var two = function(){ return 2; }

//fn(one, two);  //回调函数

fn(function(){ return 1; }, function(){ return 2; });  //匿名回调函数

4.内部函数/私有函数

在一个函数中定义另一个函数

保证了私有性

function fn(){

    var a = “a”;

    function n(){

        return a;

    }

    return n();

}

alert(fn());

alert(n());    //私有函数在外部无法访问

结果

a

没有执行

在外界可以直接执行

但是作为外层的函数 不会有有用的返回值

function fn() {

    var a = “a”;

    return function() {

alert(fn());    //返回方法体

alert(fn()());    //fn()的返回值是方法体 再再后面加括号进行调用

二、闭包

1.作用域链

var a = “a”;

function fn() {

    var b = “b”;

    alert(a);    //a

    alert(b);    //b

    function n() {

        var c = “c”;

        alert(a);    //a

        alert(b);    //b

        alert(c);    //c

    }

    return n();

}

fn();

alert(a);  //a

alert(b);  //没有执行

n();  //没有执行

外层函数将其访问权限传递给私有函数 这就是作用域链

开放私有函数访问权限

让私有函数能在外部调用

var n;

function fn() {

    n = function(){

    }

}

2.闭包结构

指的是词法表示包括不被计算的变量的函数 函数可以使用函数之外定义的变量

闭包结构的作用

开发时经常要实现这种比较茅盾的需求

要保证代码的安全性 所以使用了局部变量和私有函数

局部变量或者私有函数还需要被其他代码所用

这种需求就是茅盾的

闭包结构就解决了这个问题

闭包的问题

实现闭包的函数还是必须要定义在指定函数中的 耦合性还是不够高

 

循环中的闭包

function f() {

    var a = [];

    var i;

    for(i = 0; i < 3; i++) {

        a[i] = function() {

            return i;

        }

    }

    return a;

}

var fun = f();

alert(fun[0]());    //3

alert(fun[1]());    //3

alert(fun[2]());    //3

怎么输出0,1,2

function f() {

    var a = [];

    var i;

    for(i = 0; i < 3; i++) {

         a[i] = (

              function() {

                  return i;

              })();

    }

    return a;

}

var fun = f();

alert(fun[0]);    //0

alert(fun[1]);    //1

alert(fun[2]);    //2

或者

function f() {

    var a = [];

    var i;

    function n(x) {

        return x;

    };

    for(i = 0; i < 3; i++) {

        a[i] = n(i);

    }

    return a;

}

var fun = f();

alert(fun[0]); //0

alert(fun[1]); //1

alert(fun[2]); //2
