extjs4 - Model

{ post:time }
一、创建Model类

1.普通方式创建

(function(){

Ext.onReady(function(){

Ext.define('User', {

extend: 'Ext.data.Model',

field: [

{ name: 'name', type: 'auto' },

{ name: 'age', type: 'int' }

]

});

});

})()

2.利用Ext.regModel创建模型

这种方式可以省去extend

Ext.regModel('User', {

fields: [

{ name: 'name', type: 'auto' },

{ name: 'age', type: 'int' }

]

});

 

二、实例化模型

1.new

var u = new User({

name: 'yutou',

age: 20

});

alert(u.get('age'));

2.create方法

var u = Ext.create('User', {

name: 'yutou',

age: 20

});

3.使用ModelManager

var u = Ext.ModelMgr.create({

name: 'yutou',

age: 20

}, 'User');

创建没有数据的Model类

var u = Ext.ModelMgr.getModel('User'); 

三、数据验证

1.数据验证

Ext.regModel('User', {

fields: [

{ name: 'name', type: 'auto' },

{ name: 'age', type: 'int' }

],

validations: [

{field: 'name', type: 'length', min: 2, max: 6}

]

});

var u = Ext.create('User', {

name: 'yutou',

age: 19

});

Ext.data.validations.lengthMessage = '用户名长度错';

var errors = u.validate();

errors.each(function(error) {

alert(error.field + ' ' + error.message);

});

2.自定义数据验证

Ext.apply(Ext.data.vilidations, {

验证名称: function(config, value) {

//config 传入的是验证参数 在Model中定义验证信息的时候带的参数 例如 min max 在这里使用config.min config.max即可接收

//value 待验证的值

this.验证名称Message = this.验证名称Message + ...;  //动态修改错误信息

四、数据代理

来完成数据CRUD用的

Ext.regModel('User', {

fields: [

{ name: 'name', type: 'auto' },

{ name: 'age', type: 'int' }

],

proxy: {

type: 'ajax',

url: '....jsp'  //它响应一个json串

}

});

var u = Ext.ModelMgr.getModel('User');  //没有数据的 实例化好的对象

//使用ajax请求数据

u.load(id值, {  //id值会作为请求参数id 通过ajax请求发送到服务器

scope: this,

failure: function(record, operation) {

},

success: function(record, operation) {

alert(record.data.字段名);

},

callback: function(record, operation) {

}

});

这个只能拿一个数据 如果要拿一组数据 需要用到store

teacher 1=>many student

(function() {

Ext.onReady(function() {

Ext.getModel('teacher', {

fields: [

{name: 'teacherId', type: 'int'},

hasMany: {  //在 one 的一方配置hasMany

name : String

}

});

fields: [

{name: 'studentId', type: 'int'},

{name: 'name', type: 'auto'},

{name: 'teacher_Id', type: 'int'}

]

});
